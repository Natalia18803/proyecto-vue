<template>
  <div>
    <h1>POKEMON</h1>
    <div>
      <input v-model="nombrePokemon" placeholder="Escribe el nombre del Pokémon" />
      <button @click="searchPokemon" :disabled="cargando">Buscar Pokémon</button>
      <button @click="fetchPokemons" :disabled="cargando">Cargar Lista de Pokémon</button>
    </div>
    <div v-if="cargando">Cargando...</div>
    <div v-else-if="error">{{ error }}</div>
    <div v-else>
      <ul>
        <li v-for="pokemon in pokemons" :key="pokemon.name">
          <img :src="pokemon.sprites.front_default" :alt="pokemon.name" />
          {{ pokemon.name }}
        </li>
      </ul>
    </div>
  </div>
</template>



<script setup>
import { ref } from 'vue'
import axios from 'axios'

const pokemons = ref([])
const cargando = ref(false)
const error = ref(null)
const nombrePokemon = ref('')

const fetchPokemons = async () => {
  cargando.value = true
  error.value = null
  try {
    const response = await axios.get('https://pokeapi.co/api/v2/pokemon?limit=999')
    const results = response.data.results
    // Obtener detalles de cada Pokémon para incluir imágenes
    const pokemonDetails = await Promise.all(
      results.map(async (pokemon) => {
        const detailResponse = await axios.get(pokemon.url)
        return detailResponse.data
      })
    )
    pokemons.value = pokemonDetails
  } catch (err) {
    error.value = 'Error al cargar los Pokémon'
    console.error(err)
  } finally {
    cargando.value = false
  }
}

const searchPokemon = async () => {
  if (!nombrePokemon.value.trim()) {
    error.value = 'Por favor, escribe un nombre de Pokémon'
    return
  }
  cargando.value = true
  error.value = null
  try {
    const response = await axios.get(https://pokeapi.co/api/v2/pokemon/${nombrePokemon.value.toLowerCase()})
    pokemons.value = [response.data] // Mostrar solo el Pokémon buscado
  } catch (err) {
    error.value = 'Pokémon no encontrado'
    console.error(err)
  } finally {
    cargando.value = false
  }
}
</script>
